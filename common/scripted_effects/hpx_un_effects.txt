UN_Member = {
    # UN Members
    if = {
        limit = {
            global_var:var_un_memberlist = {
                exists = yes
            }
        }
        set_global_variable = {
            name = var_un_memberlist
            value = 0
        }
        every_country = {
            limit = {
                has_variable = un_member
            }
            change_global_variable = {
                name = var_un_memberlist
                add = 1
            }
        }
    }
}
