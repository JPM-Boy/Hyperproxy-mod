namespace = hpx_USA_event

hpx_USA_event.1 = { # New Objective
	type = country_event
	placement = root

	event_image = { video = "gfx/event_pictures/europenorthamerica_capitalists_meeting.bk2" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/firstIntl.dds"
	title = hpx_USA_event.1.t
	desc = hpx_USA_event.1.d
	flavor = hpx_USA_event.1.f
	duration = 3

	option = {
		name = hpx_USA_event.1.a
		default_option = yes
		scope:commie_country = {
			set_variable = {
				name = US_intervention
			}
		}
	}

	option = {
		name = hpx_USA_event.1.b
	}

	immediate = {
		random_country = {
			limit = {
				AND = {
					has_law = law_type:law_council_republic
					NOT = {country_rank	= rank_value:great_power}
					NOT = {has_variable = US_intervention}
				}
			}
			save_scope_as = commie_country
		}
		every_country = {
			remove_variable = US_intervention
		}
	}
}
