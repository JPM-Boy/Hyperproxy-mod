country_can_form_power_bloc = {
	is_revolutionary = no
	trigger_if = {
		OR = {
			country_rank >= rank_value:unrecognized_regional_power
		}

		OR = {
			is_country_type = recognized
			prestige >= {
				add = {
					ordered_country = {
						order_by = prestige
						position = 0

						add = prestige
						multiply = 0.05
					}
				}
			}
		}
	}
}
