great_assimilation_campaign = {
	  is_shown = {
	  	  NOT = { has_modifier = great_assimilation_policy }
	  	  NOT = { has_modifier = assimilation_policy}
	  }
     possible = {
     	OR = {
        	infamy > 35
        	root = c:EGE
        	root = c:CHI
     	}
     }

     when_taken = {
        add_modifier = {
        	name = great_assimilation_policy
        	is_decaying = yes
        	months = 24
        }
     }

    ai_chance = {
        base = 20
        modifier = {
            trigger = { root = c:EGE }
            add = 50
        }
        modifier = {
            trigger = { root = c:CHI }
            add = 50
        }
    }
}

assimilation_campaign = {
	  is_shown = {
	  	NOT = { has_modifier = great_assimilation_policy }
	  	NOT = { has_modifier = assimilation_policy}
	  }
     possible = {
        infamy > 15
     }

     when_taken = {
        add_modifier = {
        	name = assimilation_policy
        	is_decaying = yes
        	months = 48
        }
     }

    ai_chance = {
        base = 20
    }
}
